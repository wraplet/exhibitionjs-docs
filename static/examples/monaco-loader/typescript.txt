import loader from '{{MONACO_LOADER_URL}}';
import { Exhibition } from "{{EXHIBITIONJS_URL_ESM}}";

loader.config({
  paths: {
    vs: '{{MONACO_VS_URL}}'
  },
});

await loader.init().then(monaco => {
  const defaultMonacoOptions = {
    value: "",
    minimap: {
      enabled: false,
    },
    scrollBeyondLastLine: false,
    automaticLayout: true,
  };

  Exhibition.createMultiple(document.body,
    Exhibition.getMap({
      configuration: {
        editors: {
          options: {
            monacoEditorOptions: defaultMonacoOptions,
            monaco: monaco,
          }
        }
      },
    }),
  );
});
